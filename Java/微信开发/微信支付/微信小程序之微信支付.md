##### 3.微信支付接口controller

> 要注意:支付回调通知处理或退款回调通知处理，一定不要加@RequestBody或@RequestParam注解，否则可能会回调失败

```java
@RestController
@RequestMapping("/wx/pay")
@Slf4j
public class WxPayController {

    @Autowired
    private WxPaymentService wxPaymentService;

    @Autowired
    private WxGoodsRepository wxGoodsRepository;

    @Autowired
    private WxMaService wxMaService;

    /**
     * 统一下单
     */
    @GetMapping("/unifiedOrder")
    public IBaseResult<?> unifiedOrder(@RequestParam Long goodId, @RequestParam String tradeType, @RequestParam String openId) {
        log.warn("unifiedOrder--->>>>开始");
        Map<String, String> resultMap = wxPaymentService.unifiedOrder(goodId, tradeType, openId);
        log.warn("unifiedOrder--->>>>结束");
        return IBaseResultUtils.success(resultMap);
    }

    /**
     * 支付回调通知处理
     */
    @RequestMapping("/notify/order")
    public String parseOrderNotifyResult(HttpServletRequest request) throws WxPayException, IOException {
        // 直接获取字符流
        BufferedReader reader = request.getReader();
        StringBuilder requestBody = new StringBuilder();
        String line;
        while ((line = reader.readLine()) != null) {
            requestBody.append(line);
        }
        return wxPaymentService.parseOrderNotifyResult(requestBody.toString());
    }
}
```

##### 4.统一下单实现逻辑

> 注意:这里避免重复下单逻辑，查询30分钟内的未支付订单

```java
/**
 * 统一下单
 *
 * @param goodId     订单id
 * @param tradeType  JSAPI--公众号支付、NATIVE--原生扫码支付、APP--app支付，统一下单接口trade_type的传参可参考这里
 * @param openId     微信用户唯一标识
 * @return 结果
 */
@Transactional(rollbackFor = Exception.class)
public Map<String, String> unifiedOrder(Long goodId, String tradeType, String openId) {
    UserInfoEntity user = currentUser.getCurrentUser();
    if (user == null) throw new CustomException("用户未登录!");
    // 加锁是为了下面判断重复下单
    synchronized (user.getId().toString().intern()) {
        WxGoodsEntity goodsEntity = wxGoodsRepository.findById(goodId).orElse(null);
        if (goodsEntity == null) {
            log.error("不存在商品:{}", goodId);
            throw new CustomException("不存在商品,支付失败");
        }

        // 金额转分钱
        BigDecimal totalFee = goodsEntity.getTotalFee();
        long totalFeeCent = totalFee.multiply(new BigDecimal("100"))
                .setScale(0, BigDecimal.ROUND_HALF_UP)
                .longValue();

        log.warn("下单商品：" + goodsEntity.toString());
        log.warn("下单ip" + IpUtils.getIp(request));

        LocalDateTime oneDayAgo = LocalDateTime.now().minusMinutes(30);
        WxOrderEntity orderEntity = wxOrderRepository.findOne((root, query, builder) -> builder.and(
                builder.equal(root.get("goodId"), goodId),
                builder.equal(root.get("userId"), user.getId()),
                builder.equal(root.get("payStatus"), PayEnum.NOTPAY.getCode()),
                builder.greaterThanOrEqualTo(root.get("payStartTime"), DateUtil.date(oneDayAgo))
        )).orElse(null);

        try {
            String prepayId;
            if (orderEntity == null) {
                // 生成订单号
                String outTradeNo = OrderGeneratorUtil.generate();
                WxPayUnifiedOrderRequest wxPayUnifiedOrderRequest = new WxPayUnifiedOrderRequest();
                wxPayUnifiedOrderRequest.setBody(goodsEntity.getGoodName());
                wxPayUnifiedOrderRequest.setOutTradeNo(outTradeNo);
                wxPayUnifiedOrderRequest.setTotalFee((int) totalFeeCent);
                wxPayUnifiedOrderRequest.setSpbillCreateIp(IpUtils.getIp(request));//当前客户端ip
                wxPayUnifiedOrderRequest.setNotifyUrl(weChatMiniappConfig.getNotifyUrl());
                wxPayUnifiedOrderRequest.setTradeType(tradeType);//JSAPI
                wxPayUnifiedOrderRequest.setOpenid(openId);
                WxPayUnifiedOrderResult result = wxPayService.unifiedOrder(wxPayUnifiedOrderRequest);
                log.warn("统一下单返回" + result);
                prepayId = result.getPrepayId();
                String errCode = result.getErrCode();
                String errorCode = result.getErrorCode();
                String resultCode = result.getResultCode();
                log.warn("errCode:{}", errCode);
                log.warn("errorCode:{}", errorCode);
                log.warn("resultCode:{}", resultCode);
                log.warn("prepayId:{}", prepayId);
                if (!resultCode.equals("SUCCESS")) throw new CustomException("创建订单失败");
                
                // 更新订单表
                WxOrderEntity order = new WxOrderEntity();
                order.setUserId(user.getId());
                order.setGoodId(goodsEntity.getId());
                order.setPayBody(goodsEntity.getGoodName());
                order.setPayTotalFee(goodsEntity.getTotalFee());
                order.setOpenId(openId);
                order.setOrderNum(outTradeNo);
                order.setPayType(tradeType);
                order.setPrepayId(prepayId);
                order.setPayStartTime(new Timestamp(System.currentTimeMillis()));
                wxOrderRepository.save(order);
            } else {
                prepayId = orderEntity.getPrepayId();
            }

            // 返回统一下单结果，拉起微信支付参数
            Map<String, String> data = new HashMap<>();
            data.put("appId", weChatMiniappConfig.getAppid());
            data.put("timeStamp", String.valueOf(System.currentTimeMillis() / 1000));
            data.put("nonceStr", WXPayUtil.generateNonceStr());
            data.put("package", "prepay_id=" + prepayId);
            data.put("signType", "MD5");
            // 生成签名
            data.put("paySign", WXPayUtil.generateSignature(data, weChatMiniappConfig.getMchKey()));
            return data;
        } catch (Exception ex) {
            ex.printStackTrace();
            log.error("统一下单调用失败:" + ex.getMessage());
            throw new CustomException("创建订单失败!");
        }
    }
}
```

##### 5.支付回调通知处理实现逻辑

```java
/**
 * 支付回调通知处理
 *
 * @param xmlData 微信消息
 */
@Override
public String parseOrderNotifyResult(String xmlData) throws WxPayException {
    log.warn("进入支付回调通知处理:{}", xmlData);
    WxPayOrderNotifyResult notifyResult = wxPayService.parseOrderNotifyResult(xmlData);
    log.warn("完成支付回调通知处理:{}", notifyResult.toString());
    self.payNotifyResult(notifyResult.getOutTradeNo());
    return WxPayNotifyResponse.success("成功");
}


/**
 * 支付回调
 * 单独摘出来是为了方便定时任务调用
 *
 * @param orderNo 系统内订单号
 */
@Transactional(rollbackFor = Exception.class)
@Override
public void payNotifyResult(String orderNo) throws WxPayException {
    synchronized (orderNo.intern()) {
        WxOrderEntity orderEntity = wxOrderRepository.findOne((root, query, builder) -> builder.and(
                builder.equal(root.get("orderNum"), orderNo)
        )).orElse(null);
        if (orderEntity == null) return;
        // 如果支付成功直接返回
        if (orderEntity.getPayStatus().equals(PayEnum.SUCCESS.getCode())) return;
        // 更新订单状态
        WxPayOrderQueryResult result = wxPayService.queryOrder(null, orderNo);
        if (result.getTradeState().equals("SUCCESS")) {
            orderEntity.setPayStatus(PayEnum.SUCCESS.getCode());
            orderEntity.setPayEndTime(new Timestamp(System.currentTimeMillis()));
        }
        wxOrderRepository.save(orderEntity);
        log.warn("完成订单更新:{}", orderEntity.toString());
    }
}
```

##### 6.小程序调用部分

```js
import request from '~/api/request';
import config from '~/config'

let isLoading = false

/**
 * 统一下单
 * @param {*} openid 用户标识
 * @param {*} goodId 商品id
 */
export function unifiedOrder(openid, goodId) {
  if (isLoading) return
  isLoading = true
  return new Promise(async (resolve, reject) => {
    console.log(contractId);
    const res = await request('/wx/pay/unifiedOrder', 'get', {
      tradeType: 'JSAPI', //小程序暂时只能是 JSAPI
      goodId: goodId,
      openId: openid,
    }, true, '支付中...')
    if (res.code != 0) {
      wx.showToast({
        title: res.msg,
      })
      reject(res)
      isLoading = false
      return
    }

    isLoading = false
    wx.requestPayment({
      timeStamp: res.data.timeStamp,
      nonceStr: res.data.nonceStr,
      package: res.data.package,
      signType: res.data.signType,
      paySign: res.data.paySign,
      success(res) {
        console.log(res);
        resolve(res)
      },
      fail(res) {
        console.log(res);
        reject(res)
      }
    })
  })
}
```
